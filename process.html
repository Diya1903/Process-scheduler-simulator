import matplotlib.pyplot as plt
import pandas as pd

# Sample data for Gantt chart
data = {'Task': ['Task 1', 'Task 2', 'Task 3'],
        'Start': [0, 10, 20],
        'End': [5, 15, 25]}

df = pd.DataFrame(data)

# Create Gantt chart
fig, ax = plt.subplots()
ax.barh(df['Task'], df['End'] - df['Start'], left=df['Start'], color='skyblue')
ax.set_xlabel('Time')
ax.set_title('Gantt Chart')

plt.show()

import matplotlib.pyplot as plt
import numpy as np

# Sample data for efficiency comparison
scheduling_algorithms = ['FCFS', 'SJN', 'RR', 'Priority']
average_waiting_times = [25, 15, 20, 12]
cpu_utilization = [60, 70, 65, 75]
turnaround_times = [65, 55, 60, 50]

# Normalize the data for better comparison visualization
normalized_waiting_times = [(max(average_waiting_times) - x) / (max(average_waiting_times) - min(average_waiting_times)) for x in average_waiting_times]
normalized_cpu_utilization = [(x - min(cpu_utilization)) / (max(cpu_utilization) - min(cpu_utilization)) for x in cpu_utilization]
normalized_turnaround_times = [(max(turnaround_times) - x) / (max(turnaround_times) - min(turnaround_times)) for x in turnaround_times]

x = np.arange(len(scheduling_algorithms))  # the label locations

fig, ax1 = plt.subplots()

# Plotting normalized average waiting time
ax1.bar(x - 0.2, normalized_waiting_times, width=0.2, label='Avg Waiting Time', color='skyblue')
# Plotting normalized CPU utilization
ax1.bar(x, normalized_cpu_utilization, width=0.2, label='CPU Utilization', color='lightgreen')
# Plotting normalized turnaround time
ax1.bar(x + 0.2, normalized_turnaround_times, width=0.2, label='Turnaround Time', color='lightcoral')

ax1.set_xlabel('Scheduling Algorithms')
ax1.set_ylabel('Normalized Efficiency')
ax1.set_title('Efficiency Comparison of Scheduling Algorithms')
ax1.set_xticks(x)
ax1.set_xticklabels(scheduling_algorithms)
ax1.legend()

plt.tight_layout()
plt.show()
